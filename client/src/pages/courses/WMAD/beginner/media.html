Exploring Media Elements

Image Elements


<!-- Basic Image -->
<img src="image.jpg" 
     alt="Description">

<!-- Responsive Image -->
<img src="responsive.jpg" 
     alt="Responsive" 
     width="100%" 
     height="auto">

<!-- Image with Caption -->
<figure>
    <img src="photo.jpg" alt="Photo">
    <figcaption>Image Caption</figcaption>
</figure>

Video Elements


<!-- Basic Video -->
<video controls>
    <source src="video.mp4" type="video/mp4">
    Your browser does not support video.
</video>

<!-- Advanced Video -->
<video 
    src="video.mp4" 
    controls 
    width="500" 
    poster="thumbnail.jpg">
    Your browser does not support video.
</video>

Basic Video

Audio Elements


<!-- Basic Audio -->
<audio controls>
    <source src="audio.mp3" type="audio/mpeg">
    Your browser does not support audio.
</audio>

<!-- Multiple Audio Sources -->
<audio controls>
    <source src="audio.mp3" type="audio/mpeg">
    <source src="audio.ogg" type="audio/ogg">
    Your browser does not support audio.
</audio>

SVG and Canvas


<!-- Inline SVG -->
<svg width="100" height="100">
    <circle cx="50" cy="50" r="40" 
            fill="green" />
</svg>

<!-- Canvas Drawing -->
<canvas id="myCanvas" 
        width="200" 
        height="100">
    Canvas not supported.
</canvas>
